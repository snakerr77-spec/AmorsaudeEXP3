rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn() {
      return request.auth != null;
    }

    function userPath() {
      return /databases/$(database)/documents/usuarios/$(request.auth.uid);
    }

    function isOwner(userId) {
      return signedIn() && request.auth.uid == userId;
    }

    function userExists() {
      return signedIn() && exists(userPath());
    }

    function userData() {
      return get(userPath()).data;
    }

    function isActive() {
      return userExists() && userData().ativo == true;
    }

    function hasRole(roles) {
      return isActive() && userData().nivelAcesso in roles;
    }

    match /usuarios/{userId} {
      allow read: if isOwner(userId) || hasRole(["admin"]);

      allow create: if isOwner(userId)
        && request.resource.data.email == request.auth.token.email
        && request.resource.data.nivelAcesso == "colaborador"
        && request.resource.data.ativo == false;

      allow update: if hasRole(["admin"])
        || (
          isOwner(userId)
          && request.resource.data.email == resource.data.email
          && request.resource.data.nivelAcesso == resource.data.nivelAcesso
          && request.resource.data.ativo == resource.data.ativo
          && request.resource.data.diff(resource.data).affectedKeys().hasOnly([
            "nomeCompleto",
            "telefone",
            "cargo",
            "setor",
            "bio",
            "cidade",
            "foto",
            "capa",
            "ultimoLogin",
            "updatedAt"
          ])
        );

      allow delete: if hasRole(["admin"]);
    }

    match /medicos/{docId} {
      allow read: if isActive();
      allow create, update, delete: if hasRole(["admin", "financeiro", "recepcao"]);
    }

    match /prontuariosMedicos/{docId} {
      allow read, create, update, delete: if hasRole(["admin", "financeiro", "medico"]);
    }

    match /candidatosMedicos/{docId} {
      // Página pública de cadastro médico: qualquer médico pode enviar a ficha.
      // Leitura e análise ficam presas em admin/financeiro.
      allow create: if request.resource.data.status == "novo"
        && request.resource.data.keys().hasOnly([
          "name",
          "crm",
          "email",
          "phone",
          "city",
          "specialty",
          "availability",
          "payment",
          "documents",
          "experience",
          "status",
          "createdAt",
          "createdAtISO"
        ]);
      allow read, update, delete: if hasRole(["admin", "financeiro"]);
    }

    match /especialidades/{docId} {
      allow read: if isActive();
      allow create, update, delete: if hasRole(["admin", "recepcao"]);
    }

    match /horariosConsulta/{docId} {
      allow read: if isActive();
      allow create, update, delete: if hasRole(["admin", "recepcao", "medico"]);
    }

    match /exames/{docId} {
      allow read: if isActive();
      allow create, update, delete: if hasRole(["admin", "recepcao"]);
    }

    match /noticias/{docId} {
      allow read: if isActive();
      allow create, update, delete: if hasRole(["admin"]);
    }

    match /notificacoes/{docId} {
      allow read: if isActive();
      allow create, update, delete: if hasRole(["admin"]);
    }

    match /financeiro/{docId} {
      allow read, create, update, delete: if hasRole(["admin", "financeiro"]);
    }

    match /auditoria/{docId} {
      allow create: if isActive();
      allow read: if hasRole(["admin"]);
      allow update, delete: if false;
    }

    match /{document=**} {
      allow read, write: if false;
    }
  }
}
